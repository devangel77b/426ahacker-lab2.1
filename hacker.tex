\section{Introduction}
\begin{equation}
\sum\vec{F} = m\vec{a}
\end{equation}

\begin{align}
T - m_1 g \mu &= m_1 a \\
m_2 g - T &= m_2 a 
\end{align}

\begin{equation}
a = \dfrac{m_2}{m_1+m_2} g - \dfrac{m_1}{m_1+m_2} \mu g
\end{equation}
Defining $a = g \hat{x}$ and $\hat{m} = \dfrac{m_2}{m_1+m_2}$
\begin{equation}
\hat{a} = \hat{m} - (1-\hat{m}) \mu
\end{equation}
\begin{equation}
\hat{a} = (1+\mu) \hat{m} - \mu
\end{equation}

\begin{equation}
\hat{a} = 
\begin{cases}
(1+\mu) \hat{m} - \mu & \hat{m} \geq \dfrac{\mu}{1+\mu} \\
0 & \hat{m} < \dfrac{\mu}{1+\mu} 
\end{cases}
\end{equation}
